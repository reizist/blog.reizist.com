  • 東証株価指数33業種 ごとの企業数
SELECT
  base.*,
  pure.topix_33
FROM (
  SELECT
    topix_33_code,
    COUNT(*) AS count
  FROM
    `reizist-stocks.market.companies`
  GROUP BY
    topix_33_code) base
INNER JOIN (
  SELECT
    DISTINCT(topix_33),
    topix_33_code
  FROM
    reizist-stocks.market.companies) pure
ON
  base.topix_33_code = pure.topix_33_code
ORDER BY
  count DESC
